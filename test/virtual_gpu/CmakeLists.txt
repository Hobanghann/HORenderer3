message("\n######## Start to process VirtualGPU Test. ########\n")

##########################################################
# Set Test's Name
##########################################################
set(VIRTUAL_GPU_TEST_NAME virtual_gpu_tests)

##########################################################
# Set Test Suite Prefix
##########################################################
set(VIRTUAL_GPU_PREFIX VirtualGPU.)

################################################################################
# Read source files
################################################################################
file(GLOB_RECURSE SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.cc)
file(GLOB_RECURSE HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

set(ALL_FILES ${SOURCES} ${HEADERS})

################################################################################
# Add excutable target
################################################################################
include(GoogleTest)
add_executable(${VIRTUAL_GPU_TEST_NAME} ${ALL_FILES})

################################################################################
# Set target properteis
################################################################################
target_link_libraries(${VIRTUAL_GPU_TEST_NAME} PRIVATE warning_flags ${THIRD_PARTY_NAME} ${CORE_NAME} ${SHADER_NAME} ${VIRTUAL_GPU_NAME} GTest::gtest_main)
target_include_directories(${VIRTUAL_GPU_TEST_NAME} PRIVATE ${CMAKE_SOURCE_DIR})
set_target_properties(${VIRTUAL_GPU_TEST_NAME} PROPERTIES FOLDER "Test/virtual_gpu")

################################################################################
# Add test
################################################################################
gtest_discover_tests(${VIRTUAL_GPU_TEST_NAME}
  DISCOVERY_TIMEOUT 30
  TEST_PREFIX ${VIRTUAL_GPU_PREFIX}
  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
)

message("\n######## Processing VirtualGPU Test finished. ########\n")
