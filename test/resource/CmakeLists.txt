message("\n######## Start to process Resource Test. ########\n")

##########################################################
# Set Test's Name
##########################################################
set(RESOURCE_TEST_NAME resource_tests)

##########################################################
# Set Test Suite Prefix
##########################################################
set(RESOURCE_PREFIX Resource.)

################################################################################
# Read source files
################################################################################
file(GLOB_RECURSE SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.cc)
file(GLOB_RECURSE HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

set(ALL_FILES ${SOURCES} ${HEADERS})

################################################################################
# Add excutable target
################################################################################
include(GoogleTest)
add_executable(${RESOURCE_TEST_NAME} ${ALL_FILES})

################################################################################
# Set target properteis
################################################################################
target_include_directories(
  ${RESOURCE_TEST_NAME}
  PUBLIC
  ${CMAKE_CURRENT_SOURCE_DIR}
)
target_link_libraries(${RESOURCE_TEST_NAME} PRIVATE warning_flags ${THIRD_PARTY_NAME} ${CORE_NAME} ${RESOURCE_NAME} GTest::gtest_main)
set_target_properties(${RESOURCE_TEST_NAME} PROPERTIES FOLDER "Test/resource")

################################################################################
# Add test
################################################################################
gtest_discover_tests(${RESOURCE_TEST_NAME}
  DISCOVERY_TIMEOUT 30
  TEST_PREFIX ${RESOURCE_PREFIX}
  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
)

message("\n######## Processing Resource Test finished. ########\n")
