################################################################################
# Read source files
################################################################################
file(GLOB_RECURSE SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.cc)
file(GLOB_RECURSE HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

set(ALL_FILES ${SOURCES} ${HEADERS})

################################################################################
# Add excutable target
################################################################################
include(GoogleTest)
add_executable(${MATH_TEST_NAME} ${ALL_FILES})

################################################################################
# Set target properteis
################################################################################
target_link_libraries(${MATH_TEST_NAME} PRIVATE warning_flags ${THIRD_PARTY_NAME} ${CORE_NAME} GTest::gtest_main)
target_include_directories(${MATH_TEST_NAME} PRIVATE ${CMAKE_SOURCE_DIR})
set_target_properties(${MATH_TEST_NAME} PROPERTIES FOLDER "Test/core")

################################################################################
# Add test
################################################################################
gtest_discover_tests(${MATH_TEST_NAME}
  DISCOVERY_TIMEOUT 30
  TEST_PREFIX ${MATH_PREFIX}
  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
)
