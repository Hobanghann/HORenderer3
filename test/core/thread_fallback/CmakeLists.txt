################################################################################
# Read source files
################################################################################
file(GLOB_RECURSE SOURCES ${CMAKE_CURRENT_SOURCE_DIR}/*.cc)
file(GLOB_RECURSE HEADERS ${CMAKE_CURRENT_SOURCE_DIR}/*.h)

set(ALL_FILES ${SOURCES} ${HEADERS})

################################################################################
# Add excutable target
################################################################################
include(GoogleTest)
add_executable(${THREAD_FALLBACK_TEST_NAME} ${ALL_FILES})

################################################################################
# Set target properteis
################################################################################
target_include_directories(
  ${THREAD_FALLBACK_TEST_NAME}
  PUBLIC
  ${CMAKE_CURRENT_SOURCE_DIR}
)
target_link_libraries(${THREAD_FALLBACK_TEST_NAME} PRIVATE warning_flags ${THIRD_PARTY_NAME} ${CORE_NAME} GTest::gtest_main)
set_target_properties(${THREAD_FALLBACK_TEST_NAME} PROPERTIES FOLDER "Test/core")

################################################################################
# Add test
################################################################################
gtest_discover_tests(${THREAD_FALLBACK_TEST_NAME}
  DISCOVERY_TIMEOUT 30
  TEST_PREFIX ${THREAD_FALLBACK_PREFIX}
  WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
)
